DATABASE STRUCTURE

users
  ├── id (UUID)
  ├── email (string, unique)
  ├── name (string)
  ├── password_hash (string)
  ├── created_at
  └── updated_at

pools
  ├── id (UUID)
  ├── name (string)
  ├── description (text)
  ├── creator_id (FK → users)
  ├── is_public (boolean)
  ├── join_code (string, unique for private)
  ├── max_members (int)
  ├── created_at
  └── updated_at

pool_members (junction table)
  ├── id (UUID)
  ├── pool_id (FK → pools)
  ├── user_id (FK → users)
  ├── joined_at
  └── unique(pool_id, user_id)

matches
  ├── id (UUID)
  ├── match_id_external (string)
  ├── home_team (string)
  ├── away_team (string)
  ├── match_date (timestamp)
  ├── home_score (int)
  ├── away_score (int)
  ├── status (pending|completed|cancelled)
  ├── created_at
  └── updated_at

predictions
  ├── id (UUID)
  ├── pool_id (FK → pools)
  ├── user_id (FK → users)
  ├── match_id (FK → matches)
  ├── predicted_home_score (int)
  ├── predicted_away_score (int)
  ├── predicted_winner (home|away|draw)
  ├── points (int: 0, 1, or 2)
  ├── created_at
  ├── updated_at
  └── unique(pool_id, user_id, match_id)
